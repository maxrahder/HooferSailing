    }

        alert('hi');
    }
});

            alert('hi');
        }
    }

    text : 'Four',

            alert('hi');
        }
    }

    text : 'Five'
});

    alert('hi');
});

    width: 100,
    renderTo : Ext.getBody(),

    items : [ one, two, three, four, five ]
});
